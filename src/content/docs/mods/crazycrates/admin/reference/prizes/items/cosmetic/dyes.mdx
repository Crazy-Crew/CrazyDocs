<Aside type="caution">
    *Warning, The prize example assumes you know other options must be added for a Prize to function*
    *If not, I recommend you read the remaining documentation*

    *This configuration can be applied to Leather Armor, Filled Map, and other dyeable items liked Tipped Arrows*
    - Submit a suggestion if there is a dyeable item that is missing.
</Aside>

<Tabs>
    <TabCard label="Display">
        ```yml
        # The crate section.
        Crate:
          # The prizes section.
          Prizes:
            # The name of the prize.
            "1":
              DisplayItem: "potion"
              # Prize display preview settings
              Settings:
                # The rgb used for the potion.
                RGB: "229,164,229"
                # The color of the item displayed.
                # Color: "green"
        ```
    </TabCard>
    <TabCard label="Item">
        ```yml
        # The crate section.
        Crate:
          # The prizes section.
          Prizes:
            # The name of the prize.
            "1":
              DisplayItem: "potion"
              # Prize display preview settings
              Items:
                # The name of the item, good for identification internally and externally.
                "1":
                  # The item material.
                  material: "potion"
                  # Per item settings.
                  settings:
                    # The rgb of the item.
                    rgb: "229,164,229"
        ```
    </TabCard>
</Tabs>
