<Tabs>
    <TabCard label="Default">
      <Tabs>
        <TabCard label="Display">
            ```yml
            # The crate section.
            Crate:
              # The prizes section.
              Prizes:
                # The name of the prize, good for identification internally and externally.
                "1":
                  # The material of the item.
                  DisplayItem: "diamond_helmet"
                  # The patterns to display on the armor.
                  DisplayTrim:
                    # The trim material.
                    Material: "amethyst"
                    # The trim pattern.
                    Pattern: "bolt"
            ```
        </TabCard>

        <TabCard label="Item">
            ```yml
            # The crate section.
            Crate:
              # The prizes section.
              Prizes:
                # The name of the prize, good for identification internally and externally.
                "1":
                  # The material of the item.
                  DisplayItem: "diamond_helmet"
                  # Configure multiple items with whatever you want.
                  Items:
                   # The name of the item, good for identification internally and externally.
                   "1":
                     # The material of the item.
                     material: "diamond_helmet"
                     # Extra options to configure the items in this pool.
                     settings:
                       # The patterns to display on the armor.
                       trim:
                         # The trim pattern.
                         pattern: "bolt"
                         # The trim material.
                         material: "amethyst"
            ```
        </TabCard>
      </Tabs>
    </TabCard>

    <TabCard label="Legacy">
        <Aside>
            There is no "Display" under `Legacy`, because the "Display" in `Default` is sufficient.
        </Aside><br/>

        <TabCard label="Item">
            ```yml
            # The crate section.
            Crate:
              # The prizes section.
              Prizes:
                 # The name of the prize, good for identification internally and externally.
                 "1":
                   # The material of the item.
                   DisplayItem: "diamond_helmet"
                   # Configure multiple items with whatever you want.
                   Items:
                     - "item:diamond_helmet, trim:bolt!amethyst"
            ```
        </TabCard>
    </TabCard>
