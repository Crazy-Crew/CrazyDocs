<Tabs>
    <TabCard label="Default">
      <Tabs>
        <TabCard label="Display">
            ```yml
            # The voucher section.
            voucher:
              # The item material.
              item: "diamond_helmet"
              # Per item settings.
              settings:
                # The patterns to display on the shield.
                patterns:
                  base: white
                  gradient_up: light_gray
                  straight_cross: light_blue
                  flower: light_blue
                  half_horizontal: 255,0,0
            ```
        </TabCard>

        <TabCard label="Item">
            ```yml
            # The voucher section.
            voucher:
              # The item material.
              item: "diamond_helmet"
              # The items section.
              items:
                # The name of the item, good for identification internally and externally.
                "1":
                  # The item material.
                  material: "diamond_helmet"
                  # Per item settings.
                  settings:
                    # The trim to display on the helmet.
                    trim:
                      pattern: bolt
                      material: amethyst
            ```
        </TabCard>
      </Tabs>
    </TabCard>

    <TabCard label="Legacy">
        <Aside>
            There is no "Display" under `Legacy`, because the "Display" in `Default` is sufficient.
        </Aside><br/>

        <TabCard label="Item">
            ```yml
            # The voucher section.
            voucher:
              # The item material.
              item: "diamond_helmet"
              # The items section.
              items:
                - "item:diamond_helmet, trim:bolt!amethyst"
            ```
        </TabCard>
    </TabCard>
